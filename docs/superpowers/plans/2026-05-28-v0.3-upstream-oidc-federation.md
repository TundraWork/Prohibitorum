- [ ] **Step 2: Implement `pkg/federation/oidc/secret.go`**

```go
// Package oidc — secret.go
//
// AES-256-GCM encryption for upstream_idp.client_secret_enc. AAD binds the
// ciphertext to (idp_id, key_version) so copying a ciphertext between rows
// fails decryption — defense against row-swap attacks.

package oidc

import (
    "crypto/aes"
    "crypto/cipher"
    "crypto/rand"
    "fmt"
    "strconv"
)

func aadFor(idpID int64, keyVersion int32) []byte {
    return []byte("upstream_idp:" +
        strconv.FormatInt(idpID, 10) + ":" +
        strconv.Itoa(int(keyVersion)))
}

func EncryptClientSecret(dek, plaintext []byte, idpID int64, keyVersion int32) (ciphertext, nonce []byte, err error) {
    if len(dek) != 32 {
        return nil, nil, fmt.Errorf("federation/oidc: DEK must be 32 bytes (AES-256), got %d", len(dek))
    }
    block, err := aes.NewCipher(dek)
    if err != nil {
        return nil, nil, err
    }
    aead, err := cipher.NewGCM(block)
    if err != nil {
        return nil, nil, err
    }
    nonce = make([]byte, aead.NonceSize())
    if _, err := rand.Read(nonce); err != nil {
        return nil, nil, err
    }
    ciphertext = aead.Seal(nil, nonce, plaintext, aadFor(idpID, keyVersion))
    return ciphertext, nonce, nil
}

func DecryptClientSecret(dek, ciphertext, nonce []byte, idpID int64, keyVersion int32) ([]byte, error) {
    if len(dek) != 32 {
        return nil, fmt.Errorf("federation/oidc: DEK must be 32 bytes (AES-256), got %d", len(dek))
    }
    block, err := aes.NewCipher(dek)
    if err != nil {
        return nil, err
    }
    aead, err := cipher.NewGCM(block)
    if err != nil {
        return nil, err
    }
    return aead.Open(nil, nonce, ciphertext, aadFor(idpID, keyVersion))
}
```

```go
import (
    "crypto/ecdsa"
    "crypto/elliptic"
    "crypto/rand"
    "encoding/json"
    "fmt"
    "net/http"
    "net/url"
    "sync"
    "time"
)

type Server struct {
    mu         sync.Mutex
    base       string
    signingKey *ecdsa.PrivateKey
    kid        string

    // Test hooks
    nextClaims     Claims
    nextAMR        []string
    nextError      *errorParams
    issuerOverride string

    // Per-code state
    codes map[string]codeState
}

    Sub               string
```

```go
func (s *Server) handleJWKS(w http.ResponseWriter, r *http.Request) {
    pub := s.signingKey.PublicKey
    // Encode as JWK per RFC 7517. base64url of X and Y (32 bytes each for P-256).
    jwks := map[string]any{
        "keys": []map[string]any{{
            "kty": "EC",
            "crv": "P-256",
            "use": "sig",
            "alg": "ES256",
            "kid": s.kid,
            "x":   base64URLBigInt(pub.X),
            "y":   base64URLBigInt(pub.Y),
        }},
    }
    w.Header().Set("Content-Type", "application/json")
    _ = json.NewEncoder(w).Encode(jwks)
}

func base64URLBigInt(n *big.Int) string {
    b := n.Bytes()
    // Left-pad to 32 bytes for P-256.
    if len(b) < 32 {
        padded := make([]byte, 32)
        copy(padded[32-len(b):], b)
        b = padded
    }
    return base64.RawURLEncoding.EncodeToString(b)
}
```

```go
func (s *Server) handleAuthorize(w http.ResponseWriter, r *http.Request) {
    s.mu.Lock()
    if s.nextError != nil {
        errCode, desc := s.nextError.Code, s.nextError.Description
        s.nextError = nil
        s.mu.Unlock()
        redir := r.URL.Query().Get("redirect_uri")
        state := r.URL.Query().Get("state")
        u, _ := url.Parse(redir)
        q := u.Query()
        q.Set("error", errCode)
        q.Set("error_description", desc)
        q.Set("state", state)
        u.RawQuery = q.Encode()
        http.Redirect(w, r, u.String(), http.StatusFound)
        return
    }

    q := r.URL.Query()
    code := genRand(16)
    iss := s.base
    if s.issuerOverride != "" {
        iss = s.issuerOverride
    }
    s.codes[code] = codeState{
        clientID:        q.Get("client_id"),
        redirectURI:     q.Get("redirect_uri"),
        nonce:           q.Get("nonce"),
        codeChallenge:   q.Get("code_challenge"),
        challengeMethod: q.Get("code_challenge_method"),
        claims:          s.nextClaims,
        amr:             s.nextAMR,
        iss:             iss,
        expiresAt:       time.Now().Add(5 * time.Minute),
    }
    s.mu.Unlock()

    u, _ := url.Parse(q.Get("redirect_uri"))
    rq := u.Query()
    rq.Set("code", code)
    rq.Set("state", q.Get("state"))
    rq.Set("iss", iss)
    u.RawQuery = rq.Encode()
    http.Redirect(w, r, u.String(), http.StatusFound)
}

func genRand(n int) string {
    b := make([]byte, n)
    _, _ = rand.Read(b)
    return base64.RawURLEncoding.EncodeToString(b)
}
```

```go
func (s *Server) handleToken(w http.ResponseWriter, r *http.Request) {
    _ = r.ParseForm()
    code := r.Form.Get("code")
    codeVerifier := r.Form.Get("code_verifier")

    s.mu.Lock()
    state, ok := s.codes[code]
    if ok {
        delete(s.codes, code) // single-use
    }
    s.mu.Unlock()

    if !ok || time.Now().After(state.expiresAt) {
        http.Error(w, `{"error":"invalid_grant"}`, http.StatusBadRequest)
        return
    }

    // Verify PKCE.
    if state.challengeMethod != "S256" {
        http.Error(w, `{"error":"invalid_request"}`, http.StatusBadRequest)
        return
    }
    h := sha256.Sum256([]byte(codeVerifier))
    if base64.RawURLEncoding.EncodeToString(h[:]) != state.codeChallenge {
        http.Error(w, `{"error":"invalid_grant"}`, http.StatusBadRequest)
        return
    }

    // Mint ID token.
    now := time.Now()
    idClaims := map[string]any{
        "iss":                state.iss,
        "sub":                state.claims.Sub,
        "aud":                state.clientID,
        "exp":                now.Add(5 * time.Minute).Unix(),
        "iat":                now.Unix(),
        "nonce":              state.nonce,
        "email":              state.claims.Email,
        "email_verified":     state.claims.EmailVerified,
        "preferred_username": state.claims.PreferredUsername,
        "name":               state.claims.Name,
    }
    if len(state.amr) > 0 {
        idClaims["amr"] = state.amr
    }
    idToken := s.signES256(idClaims)

    resp := map[string]any{
        "access_token": genRand(24),
        "token_type":   "Bearer",
        "expires_in":   300,
        "id_token":     idToken,
        "scope":        "openid profile email",
    }
    w.Header().Set("Content-Type", "application/json")
    _ = json.NewEncoder(w).Encode(resp)
}

func (s *Server) signES256(claims map[string]any) string {
    header := map[string]any{"alg": "ES256", "typ": "JWT", "kid": s.kid}
    h, _ := json.Marshal(header)
    c, _ := json.Marshal(claims)
    signingInput := base64.RawURLEncoding.EncodeToString(h) + "." +
                    base64.RawURLEncoding.EncodeToString(c)
    digest := sha256.Sum256([]byte(signingInput))
    r, sv, _ := ecdsa.Sign(rand.Reader, s.signingKey, digest[:])
    // ECDSA-Sig per RFC 7518: r || s, each 32 bytes for P-256.
    sig := make([]byte, 64)
    rBytes := r.Bytes()
    sBytes := sv.Bytes()
    copy(sig[32-len(rBytes):32], rBytes)
    copy(sig[64-len(sBytes):], sBytes)
    return signingInput + "." + base64.RawURLEncoding.EncodeToString(sig)
}
```

```go
func (c *Client) Issuer() string        { return c.issuer }
func (c *Client) TokenEndpoint() string { return c.tokenEndpoint }

// AuthURL builds the authorization URL with PKCE S256, state, and nonce.
func (c *Client) AuthURL(state, nonce, codeChallenge string) string {
    return rp.AuthURL(state, c.rp,
        rp.WithCodeChallenge(rp.NewCodeChallenge(codeChallenge, "S256")),
        rp.WithURLParam("nonce", nonce),
    )
}

// Exchange does the token exchange + verification. expectedIss MUST equal
// the issuer claim in the returned ID token (mix-up defense). expectedNonce
// MUST equal the nonce claim. Alg allowlist applied.
func (c *Client) Exchange(ctx context.Context, code, codeVerifier, expectedIss, expectedNonce string) (*Tokens, error) {
    tokens, err := rp.CodeExchange[*zoidc.IDTokenClaims](
        ctx, code, c.rp,
        rp.WithCodeVerifier(codeVerifier),
    )
    if err != nil {
        return nil, fmt.Errorf("federation/oidc: code exchange: %w", err)
    }

    // Defense in depth: re-verify iss + nonce on top of zitadel/oidc's checks.
    if tokens.IDTokenClaims.GetIssuer() != expectedIss {
        return nil, fmt.Errorf("federation/oidc: iss mismatch: got %q expected %q",
            tokens.IDTokenClaims.GetIssuer(), expectedIss)
    }
    if tokens.IDTokenClaims.GetNonce() != expectedNonce {
        return nil, errors.New("federation/oidc: nonce mismatch")
    }
    // Defense in depth: check alg from the JOSE header is in allowlist.
    // The library should already enforce this if configured correctly, but
    // re-check explicitly.
    headerAlg := tokens.IDTokenClaims.GetSignatureAlgorithm() // placeholder; the actual API may differ
    if !slices.Contains(c.allowedAlgs, string(headerAlg)) {
        return nil, fmt.Errorf("federation/oidc: signing alg %q not in allowlist", headerAlg)
    }

    return &Tokens{
        IDToken:           tokens.IDToken,
        IDClaims:          tokens.IDTokenClaims,
        AccessToken:       tokens.AccessToken,
        AMR:               tokens.IDTokenClaims.AuthenticationMethodsReferences,
        Email:             tokens.IDTokenClaims.UserInfoEmail.Email,
        EmailVerified:     tokens.IDTokenClaims.UserInfoEmail.EmailVerified,
        PreferredUsername: tokens.IDTokenClaims.UserInfoProfile.PreferredUsername,
        Name:              tokens.IDTokenClaims.UserInfoProfile.Name,
    }, nil
}

}

}
```

```go
// Package oidc — modes.go
//
// Three provisioning modes per the master spec §"mode semantics":
//   auto_provision — create account on first sight (gated by allowed_domains + email_verified)
//   invite_only    — consume a pending enrollment with matching expected_upstream_idp_slug
//   link_only      — require an existing account_identity row; otherwise reject

package oidc

import (
    "context"
    "errors"
    "fmt"
    "slices"
    "strings"

    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgtype"

    "prohibitorum/pkg/audit"
    "prohibitorum/pkg/authn"
    "prohibitorum/pkg/db"
)

// ModesQueries is the subset of db.Querier the mode policies need.
type ModesQueries interface {
    GetAccountIdentityByIssuerSub(ctx context.Context, arg db.GetAccountIdentityByIssuerSubParams) (db.AccountIdentity, error)
    GetAccountByID(ctx context.Context, id int32) (db.Account, error)
    GetAccountByUsername(ctx context.Context, username string) (db.Account, error)
    InsertAccount(ctx context.Context, arg db.InsertAccountParams) (db.Account, error)
    InsertAccountIdentity(ctx context.Context, arg db.InsertAccountIdentityParams) (db.AccountIdentity, error)
    // For invite_only mode:
    GetEnrollmentByUpstreamIDPSlug(ctx context.Context, slug pgtype.Text) (db.Enrollment, error)
    ConsumeEnrollment(ctx context.Context, id int32) error
    // For re-login claim sync (D2):
    UpdateAccountDisplayName(ctx context.Context, arg db.UpdateAccountDisplayNameParams) error
    UpdateAccountIdentityEmail(ctx context.Context, arg db.UpdateAccountIdentityEmailParams) error
}

// Resolve dispatches per upstream_idp.mode. Returns the local account id
// and whether it was just created. Emits audit on success and failure.
func Resolve(ctx context.Context, q ModesQueries, w audit.Writer, idp *db.UpstreamIdp, claims *Tokens) (int32, bool, error) {
    upstreamIss := claims.IDClaims.GetIssuer()
    upstreamSub := claims.IDClaims.GetSubject()

    // 1. Look up existing identity.
    existing, err := q.GetAccountIdentityByIssuerSub(ctx, db.GetAccountIdentityByIssuerSubParams{
        UpstreamIss: upstreamIss, UpstreamSub: upstreamSub,
    })
    if err == nil {
        // Existing identity — sync claims, return account id.
        if err := syncClaims(ctx, q, &existing, claims); err != nil {
            // sync failure shouldn't block login.
        }
        _ = w.Record(ctx, audit.Record{
            AccountID: &existing.AccountID,
            Factor:    audit.FactorFederationOIDC,
            Event:     audit.EventUse,
            Detail:    map[string]any{"idp_slug": idp.Slug, "upstream_iss": upstreamIss},
        })
        return existing.AccountID, false, nil
    }
    if !errors.Is(err, pgx.ErrNoRows) {
        return 0, false, fmt.Errorf("federation/oidc.Resolve: get identity: %w", err)
    }

    // 2. No existing identity — dispatch by mode.
    switch idp.Mode {
    case "auto_provision":
        return applyAutoProvision(ctx, q, w, idp, claims, upstreamIss, upstreamSub)
    case "invite_only":
        return applyInviteOnly(ctx, q, w, idp, claims, upstreamIss, upstreamSub)
    case "link_only":
        return applyLinkOnly(ctx, q, w, idp, claims, upstreamIss, upstreamSub)
    default:
        return 0, false, fmt.Errorf("federation/oidc.Resolve: unknown mode %q", idp.Mode)
    }
}

func applyAutoProvision(ctx context.Context, q ModesQueries, w audit.Writer, idp *db.UpstreamIdp, claims *Tokens, iss, sub string) (int32, bool, error) {
    // D1: email_verified gating.
    if idp.RequireVerifiedEmail && !claims.EmailVerified {
        _ = w.Record(ctx, audit.Record{
            Factor: audit.FactorFederationOIDC,
            Event:  audit.EventFail,
            Detail: map[string]any{
                "idp_slug": idp.Slug, "reason": "email_not_verified",
                "upstream_iss": iss,
            },
        })
        return 0, false, authn.ErrEmailNotVerified()
    }

    // allowed_domains gating against email.
    if len(idp.AllowedDomains) > 0 && !domainAllowed(claims.Email, idp.AllowedDomains) {
        _ = w.Record(ctx, audit.Record{
            Factor: audit.FactorFederationOIDC,
            Event:  audit.EventFail,
            Detail: map[string]any{"idp_slug": idp.Slug, "reason": "domain_not_allowed"},
        })
        return 0, false, authn.ErrInviteRequired() // reuse invite_required as the "not permitted to auto-provision" hint; or add a new error
    }

    // D10: username collision check.
    username := claims.PreferredUsername
    if username == "" {
        return 0, false, fmt.Errorf("federation/oidc: upstream provided no preferred_username")
    }
    existing, err := q.GetAccountByUsername(ctx, username)
    if err == nil {
        // Local account exists with this username. We already checked identity
        // doesn't link; this is a collision.
        _ = w.Record(ctx, audit.Record{
            AccountID: &existing.ID, Factor: audit.FactorFederationOIDC, Event: audit.EventFail,
            Detail: map[string]any{"idp_slug": idp.Slug, "reason": "username_collision"},
        })
        return 0, false, authn.ErrUsernameCollision()
    }
    if !errors.Is(err, pgx.ErrNoRows) {
        return 0, false, fmt.Errorf("federation/oidc.applyAutoProvision: get username: %w", err)
    }

    // Provision new account.
    newAcct, err := q.InsertAccount(ctx, db.InsertAccountParams{
        Username:    username,
        DisplayName: claims.Name,
        Role:        "user",
        // attributes default to {} — empty JSONB
    })
    if err != nil {
        return 0, false, fmt.Errorf("federation/oidc.applyAutoProvision: insert account: %w", err)
    }

    // Link identity.
    if _, err := q.InsertAccountIdentity(ctx, db.InsertAccountIdentityParams{
        AccountID:     newAcct.ID,
        UpstreamIdpID: idp.ID,
        UpstreamIss:   iss,
        UpstreamSub:   sub,
        UpstreamEmail: pgtype.Text{String: claims.Email, Valid: claims.Email != ""},
    }); err != nil {
        return 0, false, fmt.Errorf("federation/oidc.applyAutoProvision: insert identity: %w", err)
    }

    _ = w.Record(ctx, audit.Record{
        AccountID: &newAcct.ID, Factor: audit.FactorFederationOIDC, Event: audit.EventRegister,
        Detail: map[string]any{"idp_slug": idp.Slug, "upstream_iss": iss, "reason": "auto_provision"},
    })
    _ = w.Record(ctx, audit.Record{
        AccountID: &newAcct.ID, Factor: audit.FactorFederationOIDC, Event: audit.EventUse,
        Detail: map[string]any{"idp_slug": idp.Slug, "upstream_iss": iss},
    })
    return newAcct.ID, true, nil
}

    })
    return newAcct.ID, true, nil
}
```

```go
//
// Top-level Federator orchestration. Composes secret decryption (Task 1),
// state KV (Task 4), RP client (Task 3), and mode policies (Task 4) into
// the four public flows: BeginLogin, HandleCallback, LinkBegin,
// LinkCallback.

package oidc

import (
    "context"
    "errors"

type LoginRequest struct {
    AuthorizeURL string
    StateKey     string // the random part, NOT the KV-prefixed key
}

type CallbackResult struct {

    AMR       []string
    ReturnTo  string
    IsNew     bool

type LinkResult struct {

// FederatorQueries is the subset of db.Querier the orchestrator needs.
// Concrete db.Queries satisfies this; tests inject a fake.
type FederatorQueries interface {
    ModesQueries
    GetUpstreamIDPBySlug(ctx context.Context, slug string) (db.UpstreamIdp, error)
    ListAccountIdentitiesByAccount(ctx context.Context, accountID int32) ([]db.ListAccountIdentitiesByAccountRow, error)

type Federator struct {
    q         FederatorQueries
    kvStore   kv.Store
    audit     audit.Writer
    cfg       configx.FederationConfig
    deks      map[int][]byte
    redirectURI string // computed from config.PublicOrigin + "/api/prohibitorum/auth/federation/<slug>/callback" or similar
    linkRedirectURI string
    clockNow  func() time.Time

func NewFederator(q FederatorQueries, kvStore kv.Store, w audit.Writer, cfg configx.FederationConfig, deks map[int][]byte, publicOrigin string) *Federator {
    return &Federator{
        q:               q,
        kvStore:         kvStore,
        audit:           w,
        cfg:             cfg,
        deks:            deks,
        redirectURI:     publicOrigin + "/api/prohibitorum/auth/federation/{slug}/callback",

    }

// BeginLogin starts a federation login flow. Returns the upstream authorize
// URL + the state token (caller stashes it in a cookie or returns it to the
// client).
func (f *Federator) BeginLogin(ctx context.Context, idpSlug, returnTo string) (*LoginRequest, error) {
    return f.begin(ctx, idpSlug, returnTo, nil)
}

// LinkBegin starts a link flow for an already-authenticated user. The
// callback will link the upstream identity to accountID without issuing a
// new session.
func (f *Federator) LinkBegin(ctx context.Context, accountID int32, idpSlug, returnTo string) (*LoginRequest, error) {
    return f.begin(ctx, idpSlug, returnTo, &accountID)
}

func (f *Federator) begin(ctx context.Context, idpSlug, returnTo string, linkingAccountID *int32) (*LoginRequest, error) {
    idp, err := f.q.GetUpstreamIDPBySlug(ctx, idpSlug)
    if err != nil {
        if errors.Is(err, pgx.ErrNoRows) {
            return nil, ErrUnknownIDP
        }
        return nil, fmt.Errorf("federation.BeginLogin: get idp: %w", err)
    }

    // Decrypt client secret.
    dek, ok := f.deks[int(idp.KeyVersion)]
    if !ok {
        return nil, fmt.Errorf("federation.BeginLogin: missing DEK version %d", idp.KeyVersion)
    }
    secret, err := DecryptClientSecret(dek, idp.ClientSecretEnc, idp.SecretNonce, idp.ID, idp.KeyVersion)
    if err != nil {
        return nil, fmt.Errorf("federation.BeginLogin: decrypt client_secret: %w", err)
    }

    // Build the redirectURI for this slug.
    redirectURI := f.redirectURI
    if linkingAccountID != nil {
        redirectURI = f.linkRedirectURI
    }
    redirectURI = replaceSlug(redirectURI, idpSlug)

    client, err := NewClient(ctx, idp.ClientID, string(secret), redirectURI, idp.Scopes, idp.IssuerUrl, DefaultAllowedAlgs)
    if err != nil {
        return nil, fmt.Errorf("federation.BeginLogin: new client: %w", err)
    }

    state := HelperGenerateState()
    verifier := HelperGeneratePKCEVerifier()
    nonce := HelperGenerateNonce()
    challenge := pkceChallenge(verifier) // SHA256 + base64URL

    fs := FedState{
        IDPID:                 idp.ID,
        ExpectedIss:           client.Issuer(),
        ExpectedTokenEndpoint: client.TokenEndpoint(),
        Nonce:                 nonce,
        CodeVerifier:          verifier,
        ReturnTo:              returnTo,
        LinkingAccountID:      linkingAccountID,
    }
    blob, err := fs.Encode()
    if err != nil {
        return nil, err
    }

    keyFn := LoginKey
    if linkingAccountID != nil {
        keyFn = LinkKey
    }
    if err := f.kvStore.SetEx(ctx, keyFn(state), blob, f.cfg.StateTTL); err != nil {
        return nil, fmt.Errorf("federation.BeginLogin: kv setex: %w", err)
    }

    authURL := client.AuthURL(state, nonce, challenge)
    return &LoginRequest{AuthorizeURL: authURL, StateKey: state}, nil
}

// HandleCallback consumes the state token, exchanges the code, validates
// the ID token, applies the mode policy, and returns the resolved account.
// Caller (HTTP layer) issues the session and sets the cookie.
func (f *Federator) HandleCallback(ctx context.Context, stateToken, code, issParam string) (*CallbackResult, error) {
    raw, err := f.kvStore.Pop(ctx, LoginKey(stateToken))
    if err != nil {
        return nil, authn.ErrFederationStateInvalid()
    }
    state, err := DecodeFedState(raw)
    if err != nil {
        return nil, authn.ErrFederationStateInvalid()
    }

    // RFC 9207 iss callback param defense.
    if issParam != "" && issParam != state.ExpectedIss {
        _ = f.audit.Record(ctx, audit.Record{Factor: audit.FactorFederationOIDC, Event: audit.EventFail,
            Detail: map[string]any{"reason": "iss_mismatch_callback"}})
        return nil, authn.ErrFederationStateInvalid()
    }

    // Re-fetch idp to construct the client.
    idp, err := f.q.GetUpstreamIDPBySlug(ctx, "") // implementer: how to recover slug from state? Add SlugFromState or store slug in FedState
    // ^ design note: FedState should include Slug too, OR we look up idp by ID.
    // Adding GetUpstreamIDPByID query is cleaner.
    if err != nil {
        return nil, fmt.Errorf("federation.HandleCallback: get idp: %w", err)
    }

    dek, ok := f.deks[int(idp.KeyVersion)]
    if !ok {
        return nil, fmt.Errorf("federation.HandleCallback: missing DEK")
    }
    secret, err := DecryptClientSecret(dek, idp.ClientSecretEnc, idp.SecretNonce, idp.ID, idp.KeyVersion)
    if err != nil {
        return nil, err
    }
    redirectURI := replaceSlug(f.redirectURI, idp.Slug)
    client, err := NewClient(ctx, idp.ClientID, string(secret), redirectURI, idp.Scopes, idp.IssuerUrl, DefaultAllowedAlgs)
    if err != nil {
        return nil, err
    }

    tokens, err := client.Exchange(ctx, code, state.CodeVerifier, state.ExpectedIss, state.Nonce)
    if err != nil {
        _ = f.audit.Record(ctx, audit.Record{Factor: audit.FactorFederationOIDC, Event: audit.EventFail,
            Detail: map[string]any{"reason": "code_exchange_failed", "idp_slug": idp.Slug}})
        return nil, authn.ErrFederationStateInvalid()
    }

    accountID, isNew, err := Resolve(ctx, f.q, f.audit, &idp, tokens)
    if err != nil {
        return nil, err
    }

    // D12: re-fetch account; reject if disabled.
    acct, err := f.q.GetAccountByID(ctx, accountID)
    if err != nil {
        return nil, fmt.Errorf("federation.HandleCallback: get account: %w", err)
    }
    if acct.Disabled {
        _ = f.audit.Record(ctx, audit.Record{AccountID: &accountID,
            Factor: audit.FactorFederationOIDC, Event: audit.EventFail,
            Detail: map[string]any{"reason": "account_disabled"}})
        return nil, authn.ErrBadCredentials()
    }

    return &CallbackResult{
        AccountID: accountID,
        AMR:       tokens.AMR,
        ReturnTo:  state.ReturnTo,
        IsNew:     isNew,
        IDPID:     idp.ID,
    }, nil
}

func (f *Federator) LinkCallback(ctx context.Context, stateToken, code, issParam string, currentAccountID int32) (*LinkResult, error) {
    raw, err := f.kvStore.Pop(ctx, LinkKey(stateToken))
    if err != nil {
        return nil, authn.ErrFederationStateInvalid()
    }
    state, err := DecodeFedState(raw)
    if err != nil {
        return nil, authn.ErrFederationStateInvalid()
    }
    // Session-swap defense.
    if state.LinkingAccountID == nil || *state.LinkingAccountID != currentAccountID {
        return nil, authn.ErrFederationStateInvalid()
    }
    if issParam != "" && issParam != state.ExpectedIss {
        return nil, authn.ErrFederationStateInvalid()
    }

    // Look up idp by ID (NOT slug — the slug is implicit in the URL path, but
    // the authoritative reference is the FedState's IDPID).
    idp, err := f.q.GetUpstreamIDPByID(ctx, state.IDPID)
    if err != nil {
        return nil, fmt.Errorf("federation.LinkCallback: get idp: %w", err)
    }
    dek, ok := f.deks[int(idp.KeyVersion)]
    if !ok {
        return nil, fmt.Errorf("federation.LinkCallback: missing DEK")
    }
    secret, err := DecryptClientSecret(dek, idp.ClientSecretEnc, idp.SecretNonce, idp.ID, idp.KeyVersion)
    if err != nil {
        return nil, err
    }
    redirectURI := replaceSlug(f.linkRedirectURI, idp.Slug)
    client, err := NewClient(ctx, idp.ClientID, string(secret), redirectURI, idp.Scopes, idp.IssuerUrl, DefaultAllowedAlgs)
    if err != nil {
        return nil, err
    }

    tokens, err := client.Exchange(ctx, code, state.CodeVerifier, state.ExpectedIss, state.Nonce)
    if err != nil {
        _ = f.audit.Record(ctx, audit.Record{AccountID: &currentAccountID,
            Factor: audit.FactorFederationOIDC, Event: audit.EventFail,
            Detail: map[string]any{"reason": "code_exchange_failed", "idp_slug": idp.Slug}})
        return nil, authn.ErrFederationStateInvalid()
    }

    // Link the upstream identity to the current account. The unique key
    // (upstream_iss, upstream_sub) protects against duplicate-link attempts;
    // if a different account already owns this identity, InsertAccountIdentity
    // fails with a unique violation — surface as a forbidden-link error.
    if _, err := f.q.InsertAccountIdentity(ctx, db.InsertAccountIdentityParams{
        AccountID:     currentAccountID,
        UpstreamIdpID: idp.ID,
        UpstreamIss:   tokens.IDClaims.GetIssuer(),
        UpstreamSub:   tokens.IDClaims.GetSubject(),
        UpstreamEmail: pgtype.Text{String: tokens.Email, Valid: tokens.Email != ""},
    }); err != nil {
        // Unique-violation on (upstream_iss, upstream_sub) means another
        // local account already owns this upstream identity. Don't expose
        // which account; emit a generic state-invalid error.
        _ = f.audit.Record(ctx, audit.Record{AccountID: &currentAccountID,
            Factor: audit.FactorFederationOIDC, Event: audit.EventFail,
            Detail: map[string]any{"reason": "link_conflict", "idp_slug": idp.Slug}})
        return nil, authn.ErrFederationStateInvalid()
    }

    _ = f.audit.Record(ctx, audit.Record{
        AccountID: &currentAccountID,
        Factor:    audit.FactorFederationOIDC,
        Event:     audit.EventLink,
        Detail:    map[string]any{"idp_slug": idp.Slug, "upstream_iss": tokens.IDClaims.GetIssuer()},
    })
    return &LinkResult{ReturnTo: state.ReturnTo, IDPSlug: idp.Slug}, nil
}

func replaceSlug(template, slug string) string {
    return strings.Replace(template, "{slug}", slug, 1)
}

func pkceChallenge(verifier string) string {
    h := sha256.Sum256([]byte(verifier))
    return base64.RawURLEncoding.EncodeToString(h[:])
}
```

```go
package server

// GET /api/prohibitorum/auth/federation/{slug}/login
func (s *Server) handleFederationLoginHTTP(w http.ResponseWriter, r *http.Request) {
    if s.rateLimit(w, r, "federation:ip:"+sessstore.ClientIP(r, s.config.TrustProxy), 30, time.Minute) {
        return
    }
    slug := chi.URLParam(r, "slug")
    returnTo, err := s.validateReturnTo(r.URL.Query().Get("return_to"))
    if err != nil {
        writeAuthErr(w, err)
        return
    }

    req, err := s.federator.BeginLogin(r.Context(), slug, returnTo)
    if err != nil {
        if errors.Is(err, fedoidc.ErrUnknownIDP) {
            writeAuthErr(w, authn.ErrFederationStateInvalid()) // generic-shape for unknown idp
            return
        }
        writeAuthErr(w, err)
        return
    }

    http.Redirect(w, r, req.AuthorizeURL, http.StatusFound)
}

// GET /api/prohibitorum/auth/federation/{slug}/callback
func (s *Server) handleFederationCallbackHTTP(w http.ResponseWriter, r *http.Request) {
    if s.rateLimit(w, r, "federation:ip:"+sessstore.ClientIP(r, s.config.TrustProxy), 30, time.Minute) {
        return
    }
    q := r.URL.Query()

    // Upstream error path.
    if errCode := q.Get("error"); errCode != "" {
        writeAuthErr(w, authn.ErrUpstreamError(errCode, q.Get("error_description")))
        return
    }

    state := q.Get("state")
    code := q.Get("code")
    issParam := q.Get("iss")
    if state == "" || code == "" {
        writeAuthErr(w, authn.ErrFederationStateInvalid())
        return
    }

    result, err := s.federator.HandleCallback(r.Context(), state, code, issParam)
    if err != nil {
        writeAuthErr(w, err)
        return
    }

    ip := sessstore.ClientIP(r, s.config.TrustProxy)
    ua := r.UserAgent()
    token, _, err := s.sessionStore.Issue(r.Context(), result.AccountID, ip, ua, result.AMR)
    if err != nil {
        writeAuthErr(w, err)
        return
    }

    http.SetCookie(w, sessstore.FreshSessionCookie(s.config, r, result.AccountID, token, s.config.SessionTTL))
    http.Redirect(w, r, result.ReturnTo, http.StatusFound)
}

// validateReturnTo enforces D6: empty/relative path → accepted (relative);
// absolute → must match PublicOrigin scheme+host exactly (string compare).
func (s *Server) validateReturnTo(rt string) (string, *authn.AuthError) {
    if rt == "" {
        return "/", nil
    }
    if strings.HasPrefix(rt, "/") && !strings.HasPrefix(rt, "//") {
        return rt, nil
    }
    // Absolute URL — string-compare prefix.
    if strings.HasPrefix(rt, s.config.PublicOrigin+"/") || rt == s.config.PublicOrigin {
        return rt, nil
    }
    return "", authn.ErrInvalidReturnTo()
}
```
