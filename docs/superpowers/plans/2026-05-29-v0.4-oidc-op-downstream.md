**Goal:** Ship Prohibitorum's downstream OpenID Connect Provider — Authorization Code + PKCE flow with `/authorize`, `/token` (auth_code + refresh with rotation/reuse-detection/family-revocation), `/userinfo`, `/introspect`, `/revoke`, `/oidc/logout`, real JWKS, expanded discovery, plus `signing-key` and `oidc-client` CLIs — end-to-end smoke-verified against an in-process mock RP.

```go
package oidc

import (
	"context"
	"crypto/rsa"
	"crypto/sha256"
	"encoding/base64"
	"encoding/json"
	"math/big"
	"sync"
	"time"

	"prohibitorum/pkg/db"
)

const keyCacheTTL = 5 * time.Minute

// signingKeyQueries is the subset keys.go needs.
type signingKeyQueries interface {
	ListActiveSigningKeys(ctx context.Context) ([]db.SigningKey, error)
}

type cachedKey struct {
	kid     string
	active  bool
	public  *rsa.PublicKey
	private *rsa.PrivateKey
	jwk     map[string]any
}

type keyCache struct {
	q         signingKeyQueries
	mu        sync.RWMutex
	keys      []cachedKey
	loadedAt  time.Time
	clockNow  func() time.Time
}

func newKeyCache(q signingKeyQueries) *keyCache {
	return &keyCache{q: q, clockNow: time.Now}
}

// refresh reloads keys from PG. Parses public_jwk + private_pem into crypto keys.
func (c *keyCache) refresh(ctx context.Context) error {
	rows, err := c.q.ListActiveSigningKeys(ctx)
	if err != nil {
		return err
	}
	parsed := make([]cachedKey, 0, len(rows))
	for _, r := range rows {
		priv, perr := parseRSAPrivatePEM(r.PrivatePem) // helper below
		if perr != nil {
			return perr
		}
		var jwk map[string]any
		_ = json.Unmarshal(r.PublicJwk, &jwk)
		parsed = append(parsed, cachedKey{
			kid: r.Kid, active: r.Active,
			public: &priv.PublicKey, private: priv, jwk: jwk,
		})
	}
	c.mu.Lock()
	c.keys = parsed
	c.loadedAt = c.clockNow()
	c.mu.Unlock()
	return nil
}

func (c *keyCache) maybeRefresh(ctx context.Context) {
	c.mu.RLock()
	stale := c.clockNow().Sub(c.loadedAt) > keyCacheTTL || len(c.keys) == 0
	c.mu.RUnlock()
	if stale {
		_ = c.refresh(ctx)
	}
}

func (c *keyCache) signingKey(ctx context.Context) (cachedKey, bool) {
	c.maybeRefresh(ctx)
	c.mu.RLock()
	defer c.mu.RUnlock()
	for _, k := range c.keys {
		if k.active {
			return k, true
		}
	}
	return cachedKey{}, false
}

func (c *keyCache) byKID(ctx context.Context, kid string) (cachedKey, bool) {
	c.maybeRefresh(ctx)
	c.mu.RLock()
	defer c.mu.RUnlock()
	for _, k := range c.keys {
		if k.kid == kid {
			return k, true
		}
	}
	return cachedKey{}, false
}

func (c *keyCache) jwks(ctx context.Context) map[string]any {
	c.maybeRefresh(ctx)
	c.mu.RLock()
	defer c.mu.RUnlock()
	out := make([]map[string]any, 0, len(c.keys))
	for _, k := range c.keys {
		out = append(out, k.jwk)
	}
	return map[string]any{"keys": out}
}

// publicJWK builds an RFC 7517 RSA public JWK.
func publicJWK(kid string, pub *rsa.PublicKey) map[string]any {
	return map[string]any{
		"kty": "RSA", "use": "sig", "alg": "RS256", "kid": kid,
		"n": base64.RawURLEncoding.EncodeToString(pub.N.Bytes()),
		"e": base64.RawURLEncoding.EncodeToString(big.NewInt(int64(pub.E)).Bytes()),
	}
}

// jwkThumbprint computes the RFC 7638 thumbprint of an RSA public key,
// used as the kid. Members in lexicographic order: e, kty, n.
func jwkThumbprint(pub *rsa.PublicKey) string {
	n := base64.RawURLEncoding.EncodeToString(pub.N.Bytes())
	e := base64.RawURLEncoding.EncodeToString(big.NewInt(int64(pub.E)).Bytes())
	canonical := `{"e":"` + e + `","kty":"RSA","n":"` + n + `"}`
	sum := sha256.Sum256([]byte(canonical))
	return base64.RawURLEncoding.EncodeToString(sum[:])
}
```

Add `parseRSAPrivatePEM` (PEM PKCS#8 or PKCS#1 → `*rsa.PrivateKey`) in `keys.go`:
```go
func parseRSAPrivatePEM(pemStr string) (*rsa.PrivateKey, error) {
	block, _ := pem.Decode([]byte(pemStr))
	if block == nil {
		return nil, errors.New("oidc: invalid private PEM")
	}
	if k, err := x509.ParsePKCS1PrivateKey(block.Bytes); err == nil {
		return k, nil
	}
	any, err := x509.ParsePKCS8PrivateKey(block.Bytes)
	if err != nil {
		return nil, err
	}
	rk, ok := any.(*rsa.PrivateKey)
	if !ok {
		return nil, errors.New("oidc: PEM is not an RSA private key")
	}
	return rk, nil
}
```

- [ ] **Step 2: Write `jwt.go`.**

```go
package oidc

import (
	"context"
	"errors"
	"fmt"

	"github.com/go-jose/go-jose/v4"
	"github.com/go-jose/go-jose/v4/jwt"
)

// signJWT signs claims as a compact RS256 JWS with the active key.
// typ is the JOSE "typ" header ("JWT" for id tokens, "at+jwt" for access).
func (p *Provider) signJWT(ctx context.Context, claims map[string]any, typ string) (string, error) {
	k, ok := p.keys.signingKey(ctx)
	if !ok {
		return "", errors.New("oidc: no active signing key")
	}
	signer, err := jose.NewSigner(
		jose.SigningKey{Algorithm: jose.RS256, Key: k.private},
		(&jose.SignerOptions{}).WithType(jose.ContentType(typ)).WithHeader("kid", k.kid),
	)
	if err != nil {
		return "", err
	}
	return jwt.Signed(signer).Claims(claims).Serialize()
}

// verifyJWT parses + verifies a compact RS256 JWS, resolving the key by kid.
// Rejects any alg other than RS256. Returns the raw claims map.
func (p *Provider) verifyJWT(ctx context.Context, token string) (map[string]any, error) {
	parsed, err := jwt.ParseSigned(token, []jose.SignatureAlgorithm{jose.RS256})
	if err != nil {
		return nil, fmt.Errorf("oidc: parse jwt: %w", err)
	}
	if len(parsed.Headers) != 1 {
		return nil, errors.New("oidc: unexpected JOSE header count")
	}
	kid := parsed.Headers[0].KeyID
	k, ok := p.keys.byKID(ctx, kid)
	if !ok {
		return nil, fmt.Errorf("oidc: unknown kid %q", kid)
	}
	var claims map[string]any
	if err := parsed.Claims(k.public, &claims); err != nil {
		return nil, fmt.Errorf("oidc: verify jwt: %w", err)
	}
	return claims, nil
}
```

> `jwt.ParseSigned(token, []SignatureAlgorithm{RS256})` is the go-jose/v4 alg allowlist — `alg:none`/HS256 are rejected at parse time. The `p.keys` field is the `*keyCache`, added to the `Provider` struct in Task 7's widening (for Task 1's tests, construct a `Provider{keys: newKeyCache(fake)}` directly).
