**Goal:** Land the Password + TOTP upstream authentication method on the v0.1 skeleton — including argon2id storage, RFC 6238 TOTP with AES-GCM at-rest secrets, single-use recovery codes, the two-step login flow, an extended sudo accepting any of three methods, and a `cmd/smoke` extension that exercises every new path end-to-end.

**Architecture:** Three-layer (credential store → authn flow → HTTP handlers) as established by v0.1. Cryptographic primitives are tiny in-package implementations (RFC 6238 TOTP and PHC encode/decode are ~40 lines each — too small to justify the third-party-dep blast radius). Throttle and audit are cross-cutting helpers reused across factor stores. All sensitive `/me` endpoints route through a sudo middleware that v0.2 extends from "WebAuthn only" to "any enrolled factor."

**Tech Stack:** Go 1.26, `golang.org/x/crypto/argon2`, `crypto/aes` + `crypto/cipher`, `crypto/hmac` + `crypto/sha1`, `crypto/subtle`, `encoding/base32`, pgx/v5 (via sqlc), huma v2 (existing), chi (existing).

**Spec reference:** `docs/superpowers/specs/2026-05-25-v0.2-password-totp-design.md` (delta on `2026-05-24-multi-protocol-rescope-design.md`).

```go
// Package authn — throttle.go
//
// Persistent exponential-backoff helper around auth_throttle. Used by the
// password, TOTP, and recovery-code verify paths; also by the sudo step-up
// paths for those factors. Defers all schedule decisions to configx.

package authn

import (
    "context"
    "errors"
    "time"

    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgtype"

    "prohibitorum/pkg/db"
)

// ThrottleQueries is the subset of db.Querier the throttle needs. Declared
// locally so tests can supply a fake.
type ThrottleQueries interface {
    GetAuthThrottle(ctx context.Context, arg db.GetAuthThrottleParams) (db.AuthThrottle, error)
    UpsertAuthThrottle(ctx context.Context, arg db.UpsertAuthThrottleParams) (db.AuthThrottle, error)
    ResetAuthThrottle(ctx context.Context, arg db.ResetAuthThrottleParams) error
}

type Throttle struct {
    q        ThrottleQueries
    schedule []time.Duration
    now      func() time.Time // injectable for tests
}

func NewThrottle(q ThrottleQueries, schedule []time.Duration) *Throttle {
    if len(schedule) == 0 {
        // safety net — should never happen in production, configx supplies a default
        schedule = []time.Duration{0, 0, time.Second, 5 * time.Second, 15 * time.Minute}
    }
    return &Throttle{q: q, schedule: schedule, now: time.Now}
}

// CheckLocked reports whether the (account, factor) is currently in a lockout
// window. Returns (0, nil) when free, (retryAfter, ErrFactorLocked) when locked.
func (t *Throttle) CheckLocked(ctx context.Context, accountID int32, factor string) (time.Duration, error) {
    row, err := t.q.GetAuthThrottle(ctx, db.GetAuthThrottleParams{AccountID: accountID, Factor: factor})
    if err != nil {
        if errors.Is(err, pgx.ErrNoRows) {
            return 0, nil
        }
        return 0, err
    }
    if !row.LockedUntil.Valid {
        return 0, nil
    }
    now := t.now()
    if !row.LockedUntil.Time.After(now) {
        return 0, nil
    }
    retry := row.LockedUntil.Time.Sub(now)
    return retry, ErrFactorLocked(retry)
}

// RegisterFailure increments the consecutive-failure counter and computes the
// new lockout window per the schedule. Returns the new lockout duration
// (which the caller exposes as Retry-After).
func (t *Throttle) RegisterFailure(ctx context.Context, accountID int32, factor string) (time.Duration, error) {
    row, err := t.q.GetAuthThrottle(ctx, db.GetAuthThrottleParams{AccountID: accountID, Factor: factor})
    var failed int32
    var windowStart pgtype.Timestamptz
    now := t.now()
    if err == nil {
        failed = row.FailedAttempts + 1
        windowStart = row.WindowStart
    } else if errors.Is(err, pgx.ErrNoRows) {
        failed = 1
        windowStart = pgtype.Timestamptz{Time: now, Valid: true}
    } else {
        return 0, err
    }

    idx := int(failed) - 1
    if idx >= len(t.schedule) {
        idx = len(t.schedule) - 1
    }
    lockout := t.schedule[idx]

    lockedUntil := pgtype.Timestamptz{Valid: false}
    if lockout > 0 {
        lockedUntil = pgtype.Timestamptz{Time: now.Add(lockout), Valid: true}
    }

    if _, err := t.q.UpsertAuthThrottle(ctx, db.UpsertAuthThrottleParams{
        AccountID:      accountID,
        Factor:         factor,
        FailedAttempts: failed,
        WindowStart:    windowStart,
        LockedUntil:    lockedUntil,
    }); err != nil {
        return 0, err
    }
    return lockout, nil
}

// Reset clears the throttle row on a successful verify.
func (t *Throttle) Reset(ctx context.Context, accountID int32, factor string) error {
    return t.q.ResetAuthThrottle(ctx, db.ResetAuthThrottleParams{AccountID: accountID, Factor: factor})
}
```

```go
func (w *dbWriter) Record(ctx context.Context, r Record) error {
    var detail []byte
    if r.Detail != nil {
        b, err := json.Marshal(r.Detail)
        if err != nil {
            return fmt.Errorf("audit: marshal detail: %w", err)
        }
        detail = b
    }

    var accountID pgtype.Int4
    if r.AccountID != nil {
        accountID = pgtype.Int4{Int32: *r.AccountID, Valid: true}
    }

    var credRef pgtype.Int8
    if r.CredentialRef != nil {
        credRef = pgtype.Int8{Int64: *r.CredentialRef, Valid: true}
    }

    var ip pgtype.Inet
    if r.IP != nil {
        prefix, err := netip.ParsePrefix(r.IP.String() + addrPrefix(*r.IP))
        if err != nil {
            return fmt.Errorf("audit: parse prefix: %w", err)
        }
        ip = pgtype.Inet{IPNet: prefix.Masked().Bits(), Valid: true} // adjust to pgtype.Inet's actual shape
    }
    // NOTE: the exact pgtype.Inet construction depends on the pgx version
    // shipped in go.mod. Inspect pgtype.Inet's field set:
    //   - In pgx/v5 the canonical helper is `pgtype.Inet{IPNet: net.IPNet{...}, Valid: true}`.
    //   - If the project already uses a helper (search the repo for "pgtype.Inet{"), reuse it.

    var ua pgtype.Text
    if r.UserAgent != "" {
        ua = pgtype.Text{String: r.UserAgent, Valid: true}
    }

    return w.q.InsertCredentialEvent(ctx, db.InsertCredentialEventParams{
        AccountID:     accountID,
        Factor:        string(r.Factor),
        Event:         r.Event,
        CredentialRef: credRef,
        Ip:            ip,
        UserAgent:     ua,
        Detail:        detail,
    })
}

func addrPrefix(a netip.Addr) string {
    if a.Is4() {
        return "/32"
    }
    return "/128"
}
```

```go
// Package password — phc.go
//
// PHC string encode/decode for argon2id. Format:
//   $argon2id$v=19$m=<kib>,t=<iter>,p=<lanes>$<base64 salt>$<base64 tag>
// where the base64 fields use raw (unpadded) standard encoding per the PHC spec
// (https://github.com/P-H-C/phc-string-format/blob/master/phc-sf-spec.md).

package password

import (
    "encoding/base64"
    "errors"
    "fmt"
    "strconv"
    "strings"

    "prohibitorum/pkg/configx"
)

var ErrPHCMalformed = errors.New("phc: malformed string")

type PHC struct {
    Params configx.PasswordHashParams
    Salt   []byte
    Tag    []byte
}

func PHCEncode(params configx.PasswordHashParams, salt, tag []byte) string {
    return fmt.Sprintf(
        "$argon2id$v=19$m=%d,t=%d,p=%d$%s$%s",
        params.MemoryKiB, params.Iterations, params.Parallelism,
        base64.RawStdEncoding.EncodeToString(salt),
        base64.RawStdEncoding.EncodeToString(tag),
    )
}

func PHCDecode(s string) (PHC, error) {
    // Expected segments: ["", "argon2id", "v=19", "m=...,t=...,p=...", "<b64 salt>", "<b64 tag>"]
    parts := strings.Split(s, "$")
    if len(parts) != 6 || parts[0] != "" {
        return PHC{}, ErrPHCMalformed
    }
    if parts[1] != "argon2id" {
        return PHC{}, fmt.Errorf("phc: unsupported variant %q", parts[1])
    }
    if parts[2] != "v=19" {
        return PHC{}, fmt.Errorf("phc: unsupported version %q", parts[2])
    }

    var p configx.PasswordHashParams
    for _, kv := range strings.Split(parts[3], ",") {
        eq := strings.IndexByte(kv, '=')
        if eq < 0 {
            return PHC{}, ErrPHCMalformed
        }
        key, val := kv[:eq], kv[eq+1:]
        n, err := strconv.ParseUint(val, 10, 32)
        if err != nil {
            return PHC{}, fmt.Errorf("phc: param %q: %w", key, err)
        }
        switch key {
        case "m":
            p.MemoryKiB = uint32(n)
        case "t":
            p.Iterations = uint32(n)
        case "p":
            p.Parallelism = uint8(n)
        default:
            return PHC{}, fmt.Errorf("phc: unknown param %q", key)
        }
    }
    salt, err := base64.RawStdEncoding.DecodeString(parts[4])
    if err != nil {
        return PHC{}, fmt.Errorf("phc: salt: %w", err)
    }
    tag, err := base64.RawStdEncoding.DecodeString(parts[5])
    if err != nil {
        return PHC{}, fmt.Errorf("phc: tag: %w", err)
    }
    return PHC{Params: p, Salt: salt, Tag: tag}, nil
}
```

```go
// Package password — password.go
//
// Argon2id PHC-string storage and verification per RFC 9106 + OWASP 2026
// recommended parameters (m=64MiB, t=3, p=1). Self-describing PHC format
// lets us upgrade params over time and re-hash transparently on next verify.
//
// Throttle and audit are cross-cutting helpers reused from pkg/authn and
// pkg/audit. The HTTP layer (see handle_auth_password.go) is responsible
// for the username-enumeration timing equalization — Verify here only
// handles the row-present case.

package password

import (
    "context"
    "crypto/rand"
    "crypto/subtle"
    "errors"
    "fmt"

    "golang.org/x/crypto/argon2"

    "prohibitorum/pkg/audit"
    "prohibitorum/pkg/authn"
    "prohibitorum/pkg/configx"
    "prohibitorum/pkg/db"
)

var (
    ErrPasswordNotSet    = errors.New("password: not set")
    ErrPasswordIncorrect = errors.New("password: incorrect")
)

// Tag length is fixed at 32 bytes — argon2id output sized for SHA-256
// equivalent strength.
const tagLength = 32

// Salt length is fixed at 16 bytes per OWASP recommendation.
const saltLength = 16

// PasswordQueries is the subset of db.Querier this store needs. Declared
// locally so tests can supply a fake without re-implementing all of Querier.
type PasswordQueries interface {
    GetPasswordCredential(ctx context.Context, accountID int32) (db.PasswordCredential, error)
    UpsertPasswordCredential(ctx context.Context, arg db.UpsertPasswordCredentialParams) error
    DeletePasswordCredential(ctx context.Context, accountID int32) error
}

type Store struct {
    q        PasswordQueries
    params   configx.PasswordHashParams
    throttle *authn.Throttle
    audit    audit.Writer
}

func NewStore(q PasswordQueries, params configx.PasswordHashParams, throttle *authn.Throttle, w audit.Writer) *Store {
    return &Store{q: q, params: params, throttle: throttle, audit: w}
}

// Set hashes the password with the current configx params and upserts.
func (s *Store) Set(ctx context.Context, accountID int32, password string) error {
    hash, err := s.hashPassword(password)
    if err != nil {
        return err
    }
    if err := s.q.UpsertPasswordCredential(ctx, db.UpsertPasswordCredentialParams{
        AccountID: accountID,
        Hash:      hash,
    }); err != nil {
        return fmt.Errorf("password.Set: upsert: %w", err)
    }
    _ = s.audit.Record(ctx, audit.Record{
        AccountID: &accountID,
        Factor:    audit.FactorPassword,
        Event:     audit.EventRegister,
    })
    return nil
}

// Verify confirms the supplied password matches the stored hash. Returns
// ErrPasswordNotSet when no row exists. The HTTP layer above this is
// responsible for running a dummy verify to equalize timing in that case.
func (s *Store) Verify(ctx context.Context, accountID int32, password string) error {
    row, err := s.q.GetPasswordCredential(ctx, accountID)
    if err != nil {
        return ErrPasswordNotSet
    }
    decoded, err := PHCDecode(row.Hash)
    if err != nil {
        return fmt.Errorf("password.Verify: corrupt PHC: %w", err)
    }
    candidate := argon2.IDKey(
        []byte(password), decoded.Salt,
        decoded.Params.Iterations, decoded.Params.MemoryKiB,
        decoded.Params.Parallelism, uint32(len(decoded.Tag)),
    )
    if subtle.ConstantTimeCompare(candidate, decoded.Tag) != 1 {
        _, _ = s.throttle.RegisterFailure(ctx, accountID, "password")
        _ = s.audit.Record(ctx, audit.Record{
            AccountID: &accountID,
            Factor:    audit.FactorPassword,
            Event:     audit.EventFail,
        })
        return ErrPasswordIncorrect
    }
    // Success — reset throttle, emit audit.
    _ = s.throttle.Reset(ctx, accountID, "password")
    _ = s.audit.Record(ctx, audit.Record{
        AccountID: &accountID,
        Factor:    audit.FactorPassword,
        Event:     audit.EventUse,
    })
    // Re-hash if stored params differ from current configx.
    if decoded.Params != s.params {
        newHash, err := s.hashPassword(password)
        if err == nil {
            _ = s.q.UpsertPasswordCredential(ctx, db.UpsertPasswordCredentialParams{
                AccountID: accountID,
                Hash:      newHash,
            })
        }
    }
    return nil
}

// Delete removes the password credential row. Called by
// /me/auth/revoke-password-totp.
func (s *Store) Delete(ctx context.Context, accountID int32) error {
    if err := s.q.DeletePasswordCredential(ctx, accountID); err != nil {
        return fmt.Errorf("password.Delete: %w", err)
    }
    _ = s.audit.Record(ctx, audit.Record{
        AccountID: &accountID,
        Factor:    audit.FactorPassword,
        Event:     audit.EventRevoke,
    })
    return nil
}

func (s *Store) hashPassword(password string) (string, error) {
    salt := make([]byte, saltLength)
    if _, err := rand.Read(salt); err != nil {
        return "", fmt.Errorf("password: rand salt: %w", err)
    }
    tag := argon2.IDKey(
        []byte(password), salt,
        s.params.Iterations, s.params.MemoryKiB, s.params.Parallelism, tagLength,
    )
    return PHCEncode(s.params, salt, tag), nil
}

// VerifyAgainstDummy runs an argon2id verify against a fixed dummy hash. Used
// by the HTTP layer when no row exists, to equalize timing between
// existing-but-wrong-password and missing-account.
//
// The dummy hash is computed once per Store with the current params, so it
// stays in sync with re-hash-on-upgrade.
func (s *Store) VerifyAgainstDummy(ctx context.Context, password string) {
    // We don't cache the dummy hash for v0.2 — each call hashes a constant
    // value. The cost (one argon2id) is the same as a real verify, which is
    // the entire point.
    dummySalt := make([]byte, saltLength) // all zeros — fine for a defense-only path
    _ = argon2.IDKey(
        []byte(password), dummySalt,
        s.params.Iterations, s.params.MemoryKiB, s.params.Parallelism, tagLength,
    )
}
```

```go
// Package totp — code.go
//
// RFC 6238 TOTP code computation. Step is unix_seconds / period (default 30).
// We implement HMAC-SHA1 inline since the algorithm is ~30 lines and the
// supply-chain surface of a third-party OTP library isn't worth it.

package totp

import (
    "crypto/hmac"
    "crypto/sha1"
    "encoding/binary"
    "fmt"
    "hash"
)

func computeCode(secret []byte, step int64, digits int, algorithm string) string {
    var hashFn func() hash.Hash
    switch algorithm {
    case "SHA1", "":
        hashFn = sha1.New
    // SHA256 / SHA512 are RFC 6238 options but we don't support them in v0.2.
    // The configx default is SHA1; widening is a v0.7 task.
    default:
        hashFn = sha1.New
    }
    counter := make([]byte, 8)
    binary.BigEndian.PutUint64(counter, uint64(step))
    mac := hmac.New(hashFn, secret)
    mac.Write(counter)
    digest := mac.Sum(nil)

    // Dynamic truncation per RFC 4226 §5.3.
    offset := int(digest[len(digest)-1] & 0x0f)
    code := (int64(digest[offset]&0x7f)<<24 |
        int64(digest[offset+1]&0xff)<<16 |
        int64(digest[offset+2]&0xff)<<8 |
        int64(digest[offset+3]&0xff))

    mod := int64(1)
    for i := 0; i < digits; i++ {
        mod *= 10
    }
    return fmt.Sprintf(fmt.Sprintf("%%0%dd", digits), code%mod)
}

// stepFor returns the current step value given an absolute unix time and the
// period (seconds). Exposed so the smoke test can compute codes from the
// same arithmetic.
func stepFor(unixSeconds, period int64) int64 {
    return unixSeconds / period
}
```

```go
// Package totp — aead.go
//
// AES-256-GCM encryption for TOTP secrets at rest. AAD binds the ciphertext
// to (account_id, key_version) so copying a ciphertext to a different row's
// account_id fails decryption (mitigates the cross-row swap attack).

package totp

import (
    "crypto/aes"
    "crypto/cipher"
    "crypto/rand"
    "fmt"
    "strconv"
)

func aadFor(accountID int32, keyVersion int32) []byte {
    return []byte("totp:" + strconv.Itoa(int(accountID)) + ":" + strconv.Itoa(int(keyVersion)))
}

func encryptSecret(dek, plaintext, aad []byte) (ciphertext, nonce []byte, err error) {
    if len(dek) != 32 {
        return nil, nil, fmt.Errorf("totp: DEK must be 32 bytes, got %d", len(dek))
    }
    block, err := aes.NewCipher(dek)
    if err != nil {
        return nil, nil, err
    }
    aead, err := cipher.NewGCM(block)
    if err != nil {
        return nil, nil, err
    }
    nonce = make([]byte, aead.NonceSize())
    if _, err := rand.Read(nonce); err != nil {
        return nil, nil, err
    }
    ciphertext = aead.Seal(nil, nonce, plaintext, aad)
    return ciphertext, nonce, nil
}

func decryptSecret(dek, ciphertext, nonce, aad []byte) ([]byte, error) {
    block, err := aes.NewCipher(dek)
    if err != nil {
        return nil, err
    }
    aead, err := cipher.NewGCM(block)
    if err != nil {
        return nil, err
    }
    return aead.Open(nil, nonce, ciphertext, aad)
}
```

```go
// Package totp — recovery.go
//
// Recovery codes: 16 base32 characters (uppercase A-Z + 2-7) per RFC 4648,
// formatted as XXXX-XXXX-XXXX-XXXX. 80 bits of entropy each per ANSSI's
// medium-strength threshold and the GitHub / Dropbox / Keycloak convention.
//
// Codes are argon2id-hashed at rest (PHC string format reused from
// pkg/credential/password). Verification strips hyphens, uppercases, then
// argon2id-verifies against each unused row.

package totp

import (
    "crypto/rand"
    "fmt"
    "strings"

    "prohibitorum/pkg/credential/password"
)

const (
    recoveryCodeChars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ234567" // RFC 4648 base32
    recoveryCodeLen   = 16                                  // 80 bits entropy
)

// generateRecoveryCode produces a single XXXX-XXXX-XXXX-XXXX recovery code.
func generateRecoveryCode() (string, error) {
    raw := make([]byte, recoveryCodeLen)
    if _, err := rand.Read(raw); err != nil {
        return "", fmt.Errorf("totp: recovery rand: %w", err)
    }
    chars := make([]byte, recoveryCodeLen)
    for i, b := range raw {
        chars[i] = recoveryCodeChars[int(b)%len(recoveryCodeChars)]
    }
    return fmt.Sprintf("%s-%s-%s-%s",
        chars[0:4], chars[4:8], chars[8:12], chars[12:16]), nil
}

// normalizeRecoveryCode strips hyphens + whitespace and uppercases the
// remaining characters for comparison.
func normalizeRecoveryCode(input string) string {
    s := strings.TrimSpace(input)
    s = strings.ReplaceAll(s, "-", "")
    s = strings.ToUpper(s)
    return s
}

// hashRecoveryCode argon2id-hashes the (already normalized) code with the
// configured params, returning a PHC string.
//
// We reuse the password package's PHC encode + argon2id helpers — recovery
// codes are credential-grade secrets and benefit from the same KDF.
func hashRecoveryCode(normalized string, params password.PHCParams) (string, error) {
    return password.HashRaw(normalized, params)
}

// verifyRecoveryCode argon2id-verifies normalized input against a stored
// PHC string. Returns true on match.
func verifyRecoveryCode(normalized, storedPHC string) bool {
    return password.VerifyRaw(normalized, storedPHC)
}
```

```go
// Package totp — totp.go
//
// RFC 6238 TOTP store. Manages enrollment (secret generation + AES-256-GCM
// at-rest encryption), verification (±1 step drift + last_step replay
// prevention), and the 10-recovery-code minting that happens on first
// successful verify.
//
// All cryptographic primitives are implemented in-package (code.go for
// HMAC-based code computation, aead.go for AES-GCM, recovery.go for
// recovery code format). No third-party OTP library — RFC 6238 is too
// small to justify the supply-chain surface.

package totp

import (
    "context"
    "crypto/rand"
    "encoding/base32"
    "errors"
    "fmt"
    "net/url"
    "time"

    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgtype"

    "prohibitorum/pkg/audit"
    "prohibitorum/pkg/authn"
    "prohibitorum/pkg/configx"
    "prohibitorum/pkg/credential/password"
    "prohibitorum/pkg/db"
)

var (
    ErrTOTPNotSet         = errors.New("totp: not set")
    ErrTOTPUnconfirmed    = errors.New("totp: enrollment not confirmed")
    ErrTOTPInvalidCode    = errors.New("totp: invalid code")
    ErrTOTPReplay         = errors.New("totp: code already used")
    ErrRecoveryCodeInvalid = errors.New("totp: invalid recovery code")
)

type Enrollment struct {
    SecretBase32    string
    ProvisioningURI string
    RecoveryCodes   []string // populated only when caller asks for fresh codes
}

// TOTPQueries is the subset of db.Querier this store needs.
type TOTPQueries interface {
    GetTOTPCredential(ctx context.Context, accountID int32) (db.TotpCredential, error)
    InsertTOTPCredential(ctx context.Context, arg db.InsertTOTPCredentialParams) (db.TotpCredential, error)
    DeleteTOTPCredential(ctx context.Context, accountID int32) error
    ConfirmTOTPCredential(ctx context.Context, accountID int32) error
    UpdateTOTPLastStep(ctx context.Context, arg db.UpdateTOTPLastStepParams) error
    ListRecoveryCodesByAccount(ctx context.Context, accountID int32) ([]db.RecoveryCode, error)
    InsertRecoveryCode(ctx context.Context, arg db.InsertRecoveryCodeParams) (db.RecoveryCode, error)
    ConsumeRecoveryCode(ctx context.Context, arg db.ConsumeRecoveryCodeParams) (db.RecoveryCode, error)
    DeleteAllRecoveryCodesByAccount(ctx context.Context, accountID int32) error
}

type Store struct {
    q             TOTPQueries
    deks          configx.DEKSet
    currentKeyVer int32
    cfg           configx.TOTPConfig
    throttle      *authn.Throttle
    audit         audit.Writer
    now           func() time.Time
}

func NewStore(
    q TOTPQueries,
    deks configx.DEKSet,
    cfg configx.TOTPConfig,
    throttle *authn.Throttle,
    w audit.Writer,
) *Store {
    var current int32 = 1
    for v := range deks {
        if int32(v) > current {
            current = int32(v)
        }
    }
    return &Store{q: q, deks: deks, currentKeyVer: current, cfg: cfg, throttle: throttle, audit: w, now: time.Now}
}

// Begin generates a 160-bit secret, AES-GCM encrypts under the current DEK,
// upserts (deletes any prior unconfirmed row first), and returns the
// enrollment payload. The recovery codes are NOT minted here — they're minted
// on first successful Verify.
func (s *Store) Begin(ctx context.Context, accountID int32, username string) (*Enrollment, error) {
    // 160-bit secret = 20 bytes (RFC 6238 §4 recommendation for SHA-1).
    secret := make([]byte, 20)
    if _, err := rand.Read(secret); err != nil {
        return nil, fmt.Errorf("totp.Begin: rand: %w", err)
    }

    dek, ok := s.deks[int(s.currentKeyVer)]
    if !ok {
        return nil, fmt.Errorf("totp.Begin: no DEK for version %d", s.currentKeyVer)
    }
    ct, nonce, err := encryptSecret(dek, secret, aadFor(accountID, s.currentKeyVer))
    if err != nil {
        return nil, fmt.Errorf("totp.Begin: encrypt: %w", err)
    }

    // Wipe any previous (unconfirmed or confirmed) row — caller has already
    // sudo-validated the reset path if a confirmed row existed.
    _ = s.q.DeleteTOTPCredential(ctx, accountID)
    _ = s.q.DeleteAllRecoveryCodesByAccount(ctx, accountID)

    if _, err := s.q.InsertTOTPCredential(ctx, db.InsertTOTPCredentialParams{
        AccountID:   accountID,
        SecretEnc:   ct,
        SecretNonce: nonce,
        KeyVersion:  s.currentKeyVer,
        Period:      int32(s.cfg.DefaultPeriod),
        Digits:      int32(s.cfg.DefaultDigits),
        Algorithm:   s.cfg.DefaultAlgorithm,
    }); err != nil {
        return nil, fmt.Errorf("totp.Begin: insert: %w", err)
    }

    secretB32 := base32.StdEncoding.WithPadding(base32.NoPadding).EncodeToString(secret)
    return &Enrollment{
        SecretBase32:    secretB32,
        ProvisioningURI: provisioningURI(s.cfg.Issuer, username, secretB32, s.cfg.DefaultAlgorithm, int(s.cfg.DefaultDigits), int(s.cfg.DefaultPeriod)),
    }, nil
}

// Verify checks the supplied code against the stored secret using a ±1 step
// drift window. On the first successful verify (confirmed_at NULL), it
// stamps confirmed_at and mints 10 recovery codes in the same logical
// operation. Replay (T <= last_step) is rejected with ErrTOTPReplay.
func (s *Store) Verify(ctx context.Context, accountID int32, code string) ([]string, error) {
    // Throttle gate
    if _, err := s.throttle.CheckLocked(ctx, accountID, "totp"); err != nil {
        return nil, err
    }

    row, err := s.q.GetTOTPCredential(ctx, accountID)
    if err != nil {
        if errors.Is(err, pgx.ErrNoRows) {
            return nil, ErrTOTPNotSet
        }
        return nil, err
    }

    dek, ok := s.deks[int(row.KeyVersion)]
    if !ok {
        return nil, fmt.Errorf("totp.Verify: missing DEK version %d", row.KeyVersion)
    }
    secret, err := decryptSecret(dek, row.SecretEnc, row.SecretNonce, aadFor(accountID, row.KeyVersion))
    if err != nil {
        return nil, fmt.Errorf("totp.Verify: decrypt: %w", err)
    }

    nowStep := stepFor(s.now().Unix(), int64(row.Period))
    drift := int64(s.cfg.DriftSteps)
    var matchedStep int64 = -1
    for delta := -drift; delta <= drift; delta++ {
        candidate := computeCode(secret, nowStep+delta, int(row.Digits), row.Algorithm)
        if candidate == code {
            matchedStep = nowStep + delta
            break
        }
    }
    if matchedStep < 0 {
        _, _ = s.throttle.RegisterFailure(ctx, accountID, "totp")
        _ = s.audit.Record(ctx, audit.Record{
            AccountID: &accountID, Factor: audit.FactorTOTP, Event: audit.EventFail,
        })
        return nil, ErrTOTPInvalidCode
    }
    if matchedStep <= row.LastStep {
        // Replay — counts as a failure for throttle purposes.
        _, _ = s.throttle.RegisterFailure(ctx, accountID, "totp")
        _ = s.audit.Record(ctx, audit.Record{
            AccountID: &accountID, Factor: audit.FactorTOTP, Event: audit.EventFail,
            Detail: map[string]any{"reason": "replay"},
        })
        return nil, ErrTOTPReplay
    }

    // Success path.
    if err := s.q.UpdateTOTPLastStep(ctx, db.UpdateTOTPLastStepParams{
        AccountID: accountID, LastStep: matchedStep,
    }); err != nil {
        return nil, fmt.Errorf("totp.Verify: update last_step: %w", err)
    }
    _ = s.throttle.Reset(ctx, accountID, "totp")
    _ = s.audit.Record(ctx, audit.Record{
        AccountID: &accountID, Factor: audit.FactorTOTP, Event: audit.EventUse,
    })

    if !row.ConfirmedAt.Valid {
        if err := s.q.ConfirmTOTPCredential(ctx, accountID); err != nil {
            return nil, fmt.Errorf("totp.Verify: confirm: %w", err)
        }
        codes, err := s.mintRecoveryCodes(ctx, accountID)
        if err != nil {
            return nil, fmt.Errorf("totp.Verify: mint recovery: %w", err)
        }
        return codes, nil
    }
    return nil, nil
}

// VerifyRecoveryCode argon2id-matches the supplied (XXXX-XXXX-XXXX-XXXX or
// XXXXXXXXXXXXXXXX) code against each unused recovery_code row. On match,
// marks the row used and returns nil.
func (s *Store) VerifyRecoveryCode(ctx context.Context, accountID int32, code, sessionID, ip string) error {
    if _, err := s.throttle.CheckLocked(ctx, accountID, "recovery_code"); err != nil {
        return err
    }
    normalized := normalizeRecoveryCode(code)
    if len(normalized) != recoveryCodeLen {
        _, _ = s.throttle.RegisterFailure(ctx, accountID, "recovery_code")
        _ = s.audit.Record(ctx, audit.Record{
            AccountID: &accountID, Factor: audit.FactorRecoveryCode, Event: audit.EventFail,
        })
        return ErrRecoveryCodeInvalid
    }
    rows, err := s.q.ListRecoveryCodesByAccount(ctx, accountID)
    if err != nil {
        return err
    }
    for _, row := range rows {
        if verifyRecoveryCode(normalized, row.Hash) {
            ipInet := pgtype.Inet{Valid: false}
            if ip != "" {
                // Best-effort parse — ignore failures (the row still consumes).
                if addr, parseErr := netipParseInet(ip); parseErr == nil {
                    ipInet = addr
                }
            }
            sess := pgtype.Text{String: sessionID, Valid: sessionID != ""}
            if _, err := s.q.ConsumeRecoveryCode(ctx, db.ConsumeRecoveryCodeParams{
                ID:            row.ID,
                UsedSessionID: sess,
                UsedIp:        ipInet,
            }); err != nil {
                return fmt.Errorf("totp.VerifyRecoveryCode: consume: %w", err)
            }
            _ = s.throttle.Reset(ctx, accountID, "recovery_code")
            _ = s.audit.Record(ctx, audit.Record{
                AccountID: &accountID, Factor: audit.FactorRecoveryCode, Event: audit.EventUse,
            })
            return nil
        }
    }
    _, _ = s.throttle.RegisterFailure(ctx, accountID, "recovery_code")
    _ = s.audit.Record(ctx, audit.Record{
        AccountID: &accountID, Factor: audit.FactorRecoveryCode, Event: audit.EventFail,
    })
    return ErrRecoveryCodeInvalid
}
```

```go
// HashRaw is exported for cross-package reuse (e.g. pkg/credential/totp uses
// it to hash recovery codes). Mirrors hashPassword but accepts explicit params
// to avoid coupling the caller to a particular Store instance.
func HashRaw(s string, params configx.PasswordHashParams) (string, error) {
    salt := make([]byte, saltLength)
    if _, err := rand.Read(salt); err != nil {
        return "", err
    }
    tag := argon2.IDKey(
        []byte(s), salt,
        params.Iterations, params.MemoryKiB, params.Parallelism, tagLength,
    )
    return PHCEncode(params, salt, tag), nil
}

// VerifyRaw returns true if s matches the supplied PHC string.
func VerifyRaw(s, phc string) bool {
    decoded, err := PHCDecode(phc)
    if err != nil {
        return false
    }
    candidate := argon2.IDKey(
        []byte(s), decoded.Salt,
        decoded.Params.Iterations, decoded.Params.MemoryKiB,
        decoded.Params.Parallelism, uint32(len(decoded.Tag)),
    )
    return subtle.ConstantTimeCompare(candidate, decoded.Tag) == 1
}

// DefaultParams returns the OWASP 2026 recommended argon2id parameters.
// Used by callers that don't have a Store handy (e.g. one-shot recovery
// code hashing in pkg/credential/totp).
func DefaultParams() configx.PasswordHashParams {
    return configx.PasswordHashParams{
        MemoryKiB:   65536,
        Iterations:  3,
        Parallelism: 1,
    }
}
```

```go
var ErrNoUsableMethod = errors.New("authn: account has no usable sign-in method; admin recovery required")

type Method string

const (
    MethodWebAuthn       Method = "webauthn"
    MethodPasswordTOTP   Method = "password_totp"
    MethodFederationOIDC Method = "federation_oidc"
)

// FlowQueries is the subset of db.Querier this package's flow helpers need.
type FlowQueries interface {
    ListWebauthnCredentialsByAccount(ctx context.Context, accountID int32) ([]db.WebauthnCredential, error)
    GetPasswordCredential(ctx context.Context, accountID int32) (db.PasswordCredential, error)
    GetTOTPCredential(ctx context.Context, accountID int32) (db.TotpCredential, error)
    DeletePasswordCredential(ctx context.Context, accountID int32) error
    DeleteTOTPCredential(ctx context.Context, accountID int32) error
    DeleteAllRecoveryCodesByAccount(ctx context.Context, accountID int32) error
}

func AvailableMethods(ctx context.Context, q FlowQueries, accountID int32) ([]Method, error) {
    var methods []Method

    if creds, err := q.ListWebauthnCredentialsByAccount(ctx, accountID); err == nil && len(creds) > 0 {
        methods = append(methods, MethodWebAuthn)
    } else if err != nil {
        return nil, fmt.Errorf("AvailableMethods: list webauthn: %w", err)
    }

    hasPassword := false
    if _, err := q.GetPasswordCredential(ctx, accountID); err == nil {
        hasPassword = true
    } else if !errors.Is(err, pgx.ErrNoRows) {
        return nil, fmt.Errorf("AvailableMethods: get password: %w", err)
    }
    hasConfirmedTOTP := false
    if totpRow, err := q.GetTOTPCredential(ctx, accountID); err == nil {
        if totpRow.ConfirmedAt.Valid {
            hasConfirmedTOTP = true
        }
    } else if !errors.Is(err, pgx.ErrNoRows) {
        return nil, fmt.Errorf("AvailableMethods: get totp: %w", err)
    }
    if hasPassword && hasConfirmedTOTP {
        methods = append(methods, MethodPasswordTOTP)
    }

    if len(methods) == 0 {
        return nil, ErrNoUsableMethod
    }
    return methods, nil
}

// DisableNonWebAuthnFallbacks deletes password_credential, totp_credential,
// and all recovery_code rows for the account. Called by the
// /me/auth/revoke-password-totp handler. Idempotent — deleting a non-existent
// row is a no-op at the SQL level.
//
// We don't actually wrap these in a Postgres transaction in v0.2; the three
// DELETEs are independent and partial application (e.g. password gone, totp
// still present) is recoverable by re-calling the endpoint. A future v0.3+
// hardening pass can BeginTx if the partial-failure concern grows.
func DisableNonWebAuthnFallbacks(ctx context.Context, q FlowQueries, audit audit.Writer, accountID int32) error {
    if err := q.DeletePasswordCredential(ctx, accountID); err != nil {
        return fmt.Errorf("DisableNonWebAuthnFallbacks: password: %w", err)
    }
    if err := q.DeleteTOTPCredential(ctx, accountID); err != nil {
        return fmt.Errorf("DisableNonWebAuthnFallbacks: totp: %w", err)
    }
    if err := q.DeleteAllRecoveryCodesByAccount(ctx, accountID); err != nil {
        return fmt.Errorf("DisableNonWebAuthnFallbacks: recovery: %w", err)
    }
    if audit != nil {
        for _, factor := range []string{"password", "totp", "recovery_code"} {
            _ = audit.Record(ctx, recordRevoke(accountID, factor))
        }
    }
    return nil
}

func recordRevoke(accountID int32, factor string) audit.Record {
    return audit.Record{
        AccountID: &accountID,
        Factor:    auditFactorFromString(factor),
        Event:     audit.EventRevoke,
    }
}

func auditFactorFromString(f string) audit.Factor {
    switch f {
    case "password":
        return audit.FactorPassword
    case "totp":
        return audit.FactorTOTP
    case "recovery_code":
        return audit.FactorRecoveryCode
    }
    return audit.Factor(f)
}
```

```go
// Package server — handle_auth_password.go
//
// Two-step Password+TOTP login. Step 1 (/auth/password/begin) verifies the
// password and issues an opaque partial-session token in KV (5-min TTL,
// single-use). Step 2 (/auth/totp/verify or /auth/recovery-code/verify)
// consumes the partial-session token and issues a real session on success.
//
// Username enumeration defense (D3 in the v0.2 spec): step 1 always runs
// argon2id verify even when no account / no password row exists, so timing
// is constant regardless of which branch failed.

package server

import (
    "context"
    "encoding/json"
    "errors"
    "net/http"
    "time"

    "github.com/jackc/pgx/v5"

    "prohibitorum/pkg/authn"
    "prohibitorum/pkg/credential/password"
    "prohibitorum/pkg/db"
    sessstore "prohibitorum/pkg/session"
)

type partialSession struct {
    AccountID       int32     `json:"account_id"`
    FactorCompleted string    `json:"factor_completed"`
    IssuedAt        time.Time `json:"issued_at"`
}

func partialSessionKey(token string) string {
    return "partial_session:" + token
}

// POST /api/prohibitorum/auth/password/begin
func (s *Server) handlePasswordBeginHTTP(w http.ResponseWriter, r *http.Request) {
    if s.rateLimit(w, r, "login:ip:"+sessstore.ClientIP(r, s.config.TrustProxy), 30, time.Minute) {
        return
    }
    var body struct {
        Username string `json:"username"`
        Password string `json:"password"`
    }
    if err := json.NewDecoder(r.Body).Decode(&body); err != nil || body.Username == "" || body.Password == "" {
        writeAuthErr(w, authn.ErrBadCredentials())
        return
    }

    // Look up account by username
    acct, err := s.queries.GetAccountByUsername(r.Context(), body.Username)
    if err != nil {
        if errors.Is(err, pgx.ErrNoRows) {
            // Dummy verify to equalize timing; ignore result.
            s.passwordStore.VerifyAgainstDummy(r.Context(), body.Password)
            writeAuthErr(w, authn.ErrBadCredentials())
            return
        }
        writeAuthErr(w, err)
        return
    }

    if err := s.passwordStore.Verify(r.Context(), acct.ID, body.Password); err != nil {
        if errors.Is(err, password.ErrPasswordNotSet) {
            // Dummy verify for timing parity.
            s.passwordStore.VerifyAgainstDummy(r.Context(), body.Password)
        }
        writeAuthErr(w, authn.ErrBadCredentials())
        return
    }

    // Mint a partial-session token (32 random bytes via the existing helper).
    token, err := newCeremonyToken() // reuse the same 32-byte randomness as login ceremonies
    if err != nil {
        writeAuthErr(w, err)
        return
    }
    payload, _ := json.Marshal(partialSession{
        AccountID:       acct.ID,
        FactorCompleted: "password",
        IssuedAt:        time.Now().UTC(),
    })
    if err := s.kvStore.SetEx(r.Context(), partialSessionKey(token), string(payload), s.config.Auth.PartialSessionTTL); err != nil {
        writeAuthErr(w, err)
        return
    }
    w.Header().Set("Content-Type", "application/json")
    _ = json.NewEncoder(w).Encode(map[string]string{"partial_session_token": token})
}

// POST /api/prohibitorum/auth/totp/verify
func (s *Server) handleTOTPVerifyHTTP(w http.ResponseWriter, r *http.Request) {
    if s.rateLimit(w, r, "login:ip:"+sessstore.ClientIP(r, s.config.TrustProxy), 30, time.Minute) {
        return
    }
    var body struct {
        PartialSessionToken string `json:"partial_session_token"`
        Code                string `json:"code"`
    }
    if err := json.NewDecoder(r.Body).Decode(&body); err != nil || body.PartialSessionToken == "" || body.Code == "" {
        writeAuthErr(w, authn.ErrPartialSessionInvalid())
        return
    }

    partial, err := s.consumePartialSession(r.Context(), body.PartialSessionToken)
    if err != nil {
        writeAuthErr(w, authn.ErrPartialSessionInvalid())
        return
    }

    if _, err := s.totpStore.Verify(r.Context(), partial.AccountID, body.Code); err != nil {
        writeAuthErr(w, err) // ErrTOTPInvalidCode → 401, ErrFactorLocked → 429
        return
    }

    // Issue real session with amr={pwd, otp, mfa}.
    s.issueSessionAndSetCookie(w, r, partial.AccountID, []string{"pwd", "otp", "mfa"})
}

// POST /api/prohibitorum/auth/recovery-code/verify
func (s *Server) handleRecoveryCodeVerifyHTTP(w http.ResponseWriter, r *http.Request) {
    if s.rateLimit(w, r, "login:ip:"+sessstore.ClientIP(r, s.config.TrustProxy), 30, time.Minute) {
        return
    }
    var body struct {
        PartialSessionToken string `json:"partial_session_token"`
        Code                string `json:"code"`
    }
    if err := json.NewDecoder(r.Body).Decode(&body); err != nil || body.PartialSessionToken == "" || body.Code == "" {
        writeAuthErr(w, authn.ErrPartialSessionInvalid())
        return
    }
    partial, err := s.consumePartialSession(r.Context(), body.PartialSessionToken)
    if err != nil {
        writeAuthErr(w, authn.ErrPartialSessionInvalid())
        return
    }
    ip := sessstore.ClientIP(r, s.config.TrustProxy)
    if err := s.totpStore.VerifyRecoveryCode(r.Context(), partial.AccountID, body.Code, "", ip); err != nil {
        writeAuthErr(w, err)
        return
    }
    s.issueSessionAndSetCookie(w, r, partial.AccountID, []string{"pwd", "recovery_code", "mfa"})
}

// consumePartialSession atomically reads + deletes the KV entry. Returns
// the deserialized payload on success.
func (s *Server) consumePartialSession(ctx context.Context, token string) (*partialSession, error) {
    raw, err := s.kvStore.Get(ctx, partialSessionKey(token))
    if err != nil {
        return nil, err
    }
    _ = s.kvStore.Del(ctx, partialSessionKey(token))
    var ps partialSession
    if err := json.Unmarshal([]byte(raw), &ps); err != nil {
        return nil, err
    }
    return &ps, nil
}

// issueSessionAndSetCookie wraps SessionStore.Issue + cookie set. The
// `amr` slice describes the authentication method chain for later OIDC
// ID-token claims (v0.4).
func (s *Server) issueSessionAndSetCookie(w http.ResponseWriter, r *http.Request, accountID int32, amr []string) {
    ip := sessstore.ClientIP(r, s.config.TrustProxy)
    ua := r.UserAgent()
    token, data, err := s.sessionStore.Issue(r.Context(), accountID, ip, ua, amr)
    if err != nil {
        writeAuthErr(w, err)
        return
    }
    http.SetCookie(w, sessstore.FreshSessionCookie(s.config, r, accountID, token, s.config.SessionTTL))
    _ = data
    w.WriteHeader(http.StatusNoContent)
}
```

```go
func (s *Server) stampSudoUntil(w http.ResponseWriter, r *http.Request, sess *authn.Session, method string) {
    current, _, err := s.sessionStore.Load(r.Context(), sess.Account.ID, sess.Token, sessstore.ClientIP(r, s.config.TrustProxy), r.UserAgent())
    if err != nil {
        writeAuthErr(w, err)
        return
    }
    current.SudoUntil = time.Now().Add(s.config.Auth.SudoTTL)
    if err := s.sessionStore.Save(r.Context(), sess.Account.ID, sess.Token, current); err != nil {
        writeAuthErr(w, err)
        return
    }
    _ = s.kvStore.Del(r.Context(), sudoIntentKey(sess.Data.SessionID))
    _ = s.audit.Record(r.Context(), audit.Record{
        AccountID: &sess.Account.ID,
        Factor:    audit.FactorSession,
        Event:     "sudo_granted",
        Detail:    map[string]any{"method": method},
    })
    w.WriteHeader(http.StatusNoContent)
}
```
