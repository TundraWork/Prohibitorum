# Simplified Chinese (zh) Localization Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers-extended-cc:subagent-driven-development (recommended) or superpowers-extended-cc:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a complete, high-quality Simplified Chinese (`zh`) locale to the dashboard SPA, derived from a lightly-normalized English source, with persistence + browser detection and a prod-compiler/key-parity guard covering both locales.

**Architecture:** `zh.ts` mirrors `en.ts` key-for-key. It is first scaffolded as a copy of the normalized English (so key-parity + compile guards pass immediately), then translated namespace-by-namespace against a fixed glossary, then consistency-swept. A `useLocale` composable (mirroring the existing `useTheme`) persists the choice to `localStorage` and detects `navigator.language` on first load. The `LocaleSwitcher` is already locale-agnostic and needs no change.

**Tech Stack:** Vue 3, vue-i18n (Composition mode, `legacy: false`), `@intlify/message-compiler` (prod-parity guard), vitest, vue-tsc; Go `go:embed` for the built SPA.

**Spec:** `docs/superpowers/specs/2026-06-19-cn-i18n-design.md`

## Translation Style Guide & Glossary (the contract for every translation task)

**Register:** `дҪ `, pronoun-light. Omit "you/your" where Chinese reads naturally (`з®ЎзҗҶиҙҰжҲ·`, not `з®ЎзҗҶдҪ зҡ„иҙҰжҲ·`); use `дҪ ` (never `жӮЁ`) only when a pronoun is genuinely needed.

**Rules:**
- Buttons/labels terse; descriptions are full sentences ending in `гҖӮ`. Use full-width punctuation (`пјҢгҖӮгҖҒпјҡпјҹпјҒ`) in prose; ASCII inside code/identifiers and interpolation.
- **Preserve every interpolation param verbatim:** `{count}` `{name}` `{client}` `{displayName}` `{date}` `{when}` `{app}` `{idp}` `{seconds}` `{n}` `{username}`.
- **Preserve verbatim:** CLI commands (`prohibitorum enroll-admin`), URLs, code identifiers, and any existing vue-i18n escapes вҖ” notably `{'@'}` (literal @) and the brace-escapes `{'{'}` / `{'}'}` in `admin.saml.attributeMapHint`.
- **No literal `@`** unescaped (escape as `{'@'}`), **no literal `|`** (plural separator), no unescaped `{`/`}` in any Chinese string.
- A space between CJK and a Latin token is fine (`OIDC еә”з”Ё`, `5 MB`).

```ts
/**
 * Prod-parity i18n guard, all locales.
 *
 * vue-i18n's DEV compiler only WARNS on malformed messages (e.g. a literal '@',
 * read as a linked reference); the PRODUCTION compiler THROWS at render time,
 * blanking the subtree. This compiles every leaf of every locale through
 * @intlify/message-compiler with a throwing collector вҖ” the prod path вҖ” so a
 * compile-invalid message fails here, not in a user's browser. Literal '@' must
 * be escaped as {'@'}; '{...}' interpolation and '|' plurals stay valid.
 */
import { describe, it, expect } from 'vitest'
import { baseCompile } from '@intlify/message-compiler'
import en from './en'
import zh from './zh'

function leaves(obj: unknown, path: string[] = []): Array<[string, string]> {
  if (typeof obj === 'string') return [[path.join('.'), obj]]
  if (obj && typeof obj === 'object') {
    return Object.entries(obj as Record<string, unknown>).flatMap(([k, v]) => leaves(v, [...path, k]))
  }
  return []
}

describe.each([['en', en], ['zh', zh]])('%s locale messages compile under the production compiler', (name, msgs) => {
  it(`has no message that the prod vue-i18n compiler rejects (${name})`, () => {
    const broken: string[] = []
    for (const [key, value] of leaves(msgs)) {
      const errors: string[] = []
      baseCompile(value, { onError: (e) => errors.push(`${e.code}:${e.message}`) })
      if (errors.length) broken.push(`${key} = ${JSON.stringify(value)} -> ${errors.join('; ')}`)
    }
    expect(broken, `messages rejected by the production compiler:\n${broken.join('\n')}`).toEqual([])
  })
})
```

```ts
/**
 * Key parity: zh must define exactly the same keys as en вҖ” no missing keys
 * (which would silently fall back to English) and no extra keys (dead strings).
 * This is the guard against translation drift as en evolves.
 */
import { describe, it, expect } from 'vitest'
import en from './en'
import zh from './zh'

function keys(obj: unknown, path: string[] = []): string[] {
  if (typeof obj === 'string') return [path.join('.')]
  if (obj && typeof obj === 'object') {
    return Object.entries(obj as Record<string, unknown>).flatMap(([k, v]) => keys(v, [...path, k]))
  }
  return []
}

describe('locale key parity', () => {
  it('zh defines exactly the same keys as en', () => {
    const enKeys = new Set(keys(en))
    const zhKeys = new Set(keys(zh))
    const missing = [...enKeys].filter((k) => !zhKeys.has(k))
    const extra = [...zhKeys].filter((k) => !enKeys.has(k))
    expect({ missing, extra }).toEqual({ missing: [], extra: [] })
  })
})
```

```ts
import { watch } from 'vue'
import { i18n } from '@/i18n'

const STORAGE_KEY = 'locale'

/**
 * Single source of truth for the app locale. Persists the user's choice to
 * localStorage['locale']; on first load with no stored choice, detects the
 * browser language (zh* -> 'zh', else 'en'). Call once at app start (App.vue)
 * so it applies on every route, including pre-login pages. LocaleSwitcher binds
 * the global locale directly; the watcher here persists whatever it sets.
 *
 * CSP is `script-src 'self'`, so the locale is applied during Vue bootstrap
 * (no inline pre-paint script); text reflow is instant.
 */
function resolveInitial(): string {
  const available = i18n.global.availableLocales as string[]
  const stored = localStorage.getItem(STORAGE_KEY)
  if (stored && available.includes(stored)) return stored
  const nav = (navigator.language ?? '').toLowerCase()
  if (nav.startsWith('zh')) return 'zh'
  return 'en'
}

export function useLocale() {
  const locale = i18n.global.locale // WritableComputedRef<string> (legacy: false)
  const initial = resolveInitial()
  if (locale.value !== initial) locale.value = initial
  watch(locale, (v) => {
    try { localStorage.setItem(STORAGE_KEY, v) } catch { /* storage unavailable */ }
  })
  function setLocale(l: string): void { locale.value = l }
  return { locale, setLocale }
}
```
